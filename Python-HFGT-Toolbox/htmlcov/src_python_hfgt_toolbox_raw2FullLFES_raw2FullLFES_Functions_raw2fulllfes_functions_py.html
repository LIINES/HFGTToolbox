217 ''' 

218 The function calculates the Transformation Knowledge Base JM, Constraints Matrix KM,  

219 and Transformation Concept AM. 

220 :param LFES: 

221 :return: 

222 ''' 

223 print("I am entering calcJM_KM_AM.py") 

224 for k1 in range(LFES.numMachines): 

225 idxL = lil_matrix((LFES.numTransformProcess, 1), dtype=int) 

226 if isinstance(LFES.machines.methodsxForm[k1], methodxForm): 

227 for k2 in range(len(LFES.machines.methodsxForm[k1].name)): 

228 idxTemp = np.array([1 if LFES.machines.methodsxForm[k1].name[k2] == elem else 0 for elem in LFES.setTransformProcess]).reshape(LFES.numTransformProcess, 1) 

229 idxL += idxTemp 

230 (i,j,v) = find(idxL) 

231 LFES.JM.row = np.concatenate((LFES.JM.row,i)) 

232 LFES.JM.col = np.concatenate((LFES.JM.col,np.repeat(k1,len(i)))) 

233 LFES.JM.data = np.concatenate((LFES.JM.data,v)) 

234 if not(LFES.machines.methodsxForm[k1].status[k2]=='true'): 

235 LFES.KM.row = np.concatenate((LFES.KM.row,i)) 

236 LFES.KM.col = np.concatenate((LFES.KM.col,[k1])) 

237 LFES.KM.data = np.concatenate((LFES.KM.data,v)) 

238 if LFES.KM.nnz != 0: 

239 LFES.AM = LFES.JM.tocsr()-LFES.KM.tocsr() 

240 else: 

241 LFES.AM = LFES.JM.tocsr()

1522def matricize(ATensor, R, C): 

1523 ''' 

1524 This function transforms a tensor into a 2-D matrix. 

1525 :param ATensor: tensor to be transformed 

1526 :param R: dimension indicies to be merged 

1527 :param C: dimension indicies to be maintained 

1528 :return: AMatrix: matrix form of the input tensor 

1529 ''' 

1530 print("I am entering matricize.py") 

1531 P = list(ATensor.shape) 

1532 J = np.prod([P[i] for i in R]) 

1533 K = [P[i] for i in C] 

1534 K.append(J) 

1535 R = np.array(R) 

1536 

1537 AMatrix = COO([[0]] * len(K), 1, shape=(1, 1)) 

1538 AMatrix.shape = K 

1539 AMatrix.data = ATensor.data 

1540 coords = ATensor.coords 

1541 newReshape = np.ndarray(shape=(len(K), ATensor.nnz), dtype=np.int64) 

1542 

1543 flattend = coords[R[0]] 

1544 for k1 in range(len(R) - 1): 

1545 flattend = flattend + coords[R[k1 + 1]] * P[R[k1]] 

1546 

1547 newReshape[min(R)] = flattend 

1548 K[min(R)] = J 

1549 for k2 in C: 

1550 if sum(R < k2): 

1551 K[k2 - sum(R < k2) + 1] = P[k2] 

1552 newReshape[k2 - sum(R < k2) + 1] = coords[k2] 

1553 else: 

1554 K[k2] = P[k2] 

1555 newReshape[k2] = coords[k2] 

1556 

1557 AMatrix.coords = newReshape 

1558 AMatrix.shape = tuple(K) 

1559 

1560 print("I am exiting matricize.py") 

1561 return AMatrix
